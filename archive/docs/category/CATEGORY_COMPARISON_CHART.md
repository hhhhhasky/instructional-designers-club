### 3. API实现对比

#### 修复前的代码

```typescript
export async function getCourseCategories(): Promise<string[]> {
  try {
    // ❌ 错误：从 courses 表获取
    const { data, error } = await supabase
      .from('courses')  // 错误的数据源
      .select('category')
      .eq('status', 'published');

    if (error) throw error;
    if (!Array.isArray(data)) return [];

    // ❌ 去重但顺序不稳定
    const categories = Array.from(
      new Set(data.map(item => item.category).filter(Boolean))
    );
    
    return ['全部', ...categories];
  } catch (error) {
    console.error('获取课程分类异常:', error);
    return ['全部'];
  }
}
```

**问题：**
1. ❌ 从 courses 表获取分类
2. ❌ 依赖课程实际使用情况
3. ❌ 排序不稳定
4. ❌ 空分类不显示

#### 修复后的代码

```typescript
export async function getCourseCategories(): Promise<string[]> {
  try {
    // ✅ 正确：从 course_categories 表获取
    const { data, error } = await supabase
      .from('course_categories')  // 正确的数据源
      .select('name')
      .order('sort_order', { ascending: true });  // 稳定排序

    if (error) throw error;
    if (!Array.isArray(data)) return ['全部'];

    // ✅ 提取分类名称
    const categories = data.map(item => item.name).filter(Boolean);
    
    return ['全部', ...categories];
  } catch (error) {
    console.error('获取课程分类异常:', error);
    return ['全部'];
  }
}
```

**改进：**
1. ✅ 从 course_categories 表获取分类
2. ✅ 使用标准分类定义
3. ✅ 按 sort_order 稳定排序
4. ✅ 所有分类都显示
